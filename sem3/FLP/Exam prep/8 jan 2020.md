(defun fct(k l crrk)
  (cond
    ((null l) nil)
    ((atom l)
      (cond
        ((equal crrk k) 0)
        (t l)
      )
    )
    (t (mapcar #'(lambda (x) 
          (cond
            ((listp x) (fct k x (+ crrk 1)))
            (t (fct k x crrk))
          )
        ) l 
       ) 
    )
  )
